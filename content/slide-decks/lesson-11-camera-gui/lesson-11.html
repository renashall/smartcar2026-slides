Build the file · chunk 4 / 15

connect(): open two sockets

Pictures in on 8000, commands out on 5000 — the car server's two ports.

Connection 1 — video in on port 8000.

Connection 2 — commands out on port 5000.

makefile("rb") reads an exact number of bytes.

lesson_11_camera_gui.py
def connect(pi_ip):
    """Open the video and command sockets."""
    global video_socket, video_file, command_socket
    # Connection 1: pictures come in on port 8000.
    video_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    video_socket.connect((pi_ip, VIDEO_PORT))
    video_file = video_socket.makefile("rb")  # read like a file

    # Connection 2: servo commands go out on port 5000.
    command_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    command_socket.connect((pi_ip, COMMAND_PORT))

Build the file · chunk 5 / 15

read_frame(): size first, then the picture

A 4-byte size, then that many JPEG bytes — exactly like Lesson 6.

Read 4 bytes → the picture's size.

A short or zero read → stream ended.

imdecode → image, no file on disk.

lesson_11_camera_gui.py
def read_frame():
    """Read one picture: a 4-byte size, then JPEG bytes."""
    header = video_file.read(4)            # the picture's size
    if len(header) != 4:
        return None
    length = struct.unpack("<L", header)[0]   # bytes -> number
    if length == 0:
        return None
    jpeg_data = video_file.read(length)    # exactly that many
    if len(jpeg_data) != length:
        return None
    return cv2.imdecode(np.frombuffer(jpeg_data, dtype=np.uint8),
                        cv2.IMREAD_COLOR)

Build the file · chunk 6 / 15

send_servo() & clamp()

A head command is just text: CMD_SERVO#0#95\n.

Name # channel # angle, ending in a newline.

Not connected yet? Do nothing.

clamp() keeps every angle in its safe range.

lesson_11_camera_gui.py
def send_servo(channel, angle):
    """Send one servo command, e.g. CMD_SERVO#0#95."""
    if command_socket is None:
        return                          # not connected yet
    message = cmd.CMD_SERVO + "#" + channel + "#" \
              + str(int(angle)) + "\n"
    command_socket.send(message.encode("utf-8"))


def clamp(value, low, high):
    """Keep a number between low and high."""
    return max(low, min(high, value))

Build the file · chunk 7 / 15

track_face(): nudge toward center

Offset is -1 at the edges and 0 in the middle — move a small step the right way.

How far is the face from the middle?

Inside the dead zone → don't move.

Else nudge pan & tilt by SERVO_STEP.

lesson_11_camera_gui.py
def track_face(face):
    """Turn the head so the face moves toward the middle."""
    global pan_angle, tilt_angle
    x, y, w, h = face
    # -1 at left/top, +1 at right/bottom, 0 in the middle.
    offset_x = ((x + w / 2) / FRAME_WIDTH - 0.5) * 2
    offset_y = ((y + h / 2) / FRAME_HEIGHT - 0.5) * 2
    if abs(offset_x) < DEAD_ZONE and abs(offset_y) < DEAD_ZONE:
        return                          # already centred
    pan_angle = clamp(pan_angle + SERVO_STEP * offset_x, PAN_MIN, PAN_MAX)
    tilt_angle = clamp(tilt_angle - SERVO_STEP * offset_y, TILT_MIN, TILT_MAX)
    send_servo(PAN_SERVO, pan_angle)
    send_servo(TILT_SERVO, tilt_angle)

Build the file · chunk 8 / 15

video_thread(): the background loop

The slow work runs here — that's why the buttons keep responding.

Read a frame; if tracking, box every face.

Follow the biggest with max(...).

Store it under the lock, safely.

lesson_11_camera_gui.py
def video_thread():
    """Read a picture, optionally find faces, store it."""
    global latest_frame
    while running:
        frame = read_frame()
        if frame is None:
            break
        if track_faces:                  # box faces, follow biggest
            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            faces = face_cascade.detectMultiScale(gray, 1.3, 5)
            for (x, y, w, h) in faces:
                cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 0), 2)
            if len(faces) > 0:
                track_face(max(faces, key=lambda f: f[2] * f[3]))
        with frame_lock:                 # hand it over safely
            latest_frame = frame

on_connect(): connect, then start the thread

A bad IP shows a message instead of crashing — then the background reader starts.

connect() with the IP from the box.

Bad IP → OSError → show a message.

Start a daemon thread (stops with the app).

lesson_11_camera_gui.py
    def on_connect(self):
        """Connect, then start the background video thread."""
        global running
        try:
            connect(self.ip_box.text())  # IP from the box
        except OSError as error:
            self.video_label.setText("Could not connect:\n" + str(error))
            return
        running = True
        thread = threading.Thread(target=video_thread)
        thread.daemon = True          # stop when the app closes
        thread.start()

Button handlers: toggle, nudge, home

Both the buttons and track_face() reuse send_servo() and clamp().

on_track_changed → flip the flag.

nudge → one servo, one step, clamped.

go_home → recentre to 90°.

lesson_11_camera_gui.py
    def on_track_changed(self):
        global track_faces
        track_faces = self.track_checkbox.isChecked()

    def nudge(self, channel, change):
        global pan_angle, tilt_angle
        if channel == PAN_SERVO:
            pan_angle = clamp(pan_angle + change, PAN_MIN, PAN_MAX)
            send_servo(PAN_SERVO, pan_angle)
        else:
            tilt_angle = clamp(tilt_angle + change, TILT_MIN, TILT_MAX)
            send_servo(TILT_SERVO, tilt_angle)

    def go_home(self):
        global pan_angle, tilt_angle
        pan_angle, tilt_angle = 90.0, 90.0
        send_servo(PAN_SERVO, pan_angle)
        send_servo(TILT_SERVO, tilt_angle)

refresh_video(): copy, convert, show

The colour swap (BGR → RGB) is why faces look natural, not blue-tinted.

Quick copy under the lock.

No picture yet → return.

Wrap pixels in a QImage, draw it.

lesson_11_camera_gui.py
    def refresh_video(self):
        """Copy the newest picture and draw it in the window."""
        with frame_lock:
            frame = None if latest_frame is None else latest_frame.copy()
        if frame is None:
            return                          # no picture yet
        # OpenCV is Blue-Green-Red; Qt expects Red-Green-Blue.
        rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        height, width, _ = rgb.shape
        image = QtGui.QImage(rgb.data, width, height, 3 * width,
                             QtGui.QImage.Format_RGB888)
        self.video_label.setPixmap(QtGui.QPixmap.fromImage(image))

destroy() & the main block

PyQt runs its own loop — so it's try / finally, but the clean-up habit is the same.

Stop the thread, close the sockets.

One QApplication, show the window.

finally always cleans up.

lesson_11_camera_gui.py
def destroy():
    """Stop the thread and close the connections."""
    global running
    running = False                    # tell the thread to stop
    if video_file is not None:
        video_file.close()
    if video_socket is not None:
        video_socket.close()
    if command_socket is not None:
        command_socket.close()


if __name__ == "__main__":
    app = QtWidgets.QApplication(sys.argv)   # every PyQt app needs one
    window = CameraWindow()
    window.show()
    try:
        app.exec_()                      # runs until you close it
    finally:
        destroy()                       # always clean up

Common issues

Most issues are the wrong IP or the server not started — check those first.

"Could not connect"
Wrong IP, or ports 5000/8000 blocked. Fix it and click Connect again.
Image is black or frozen
Check the camera cable; restart the GUI and the Pi server.
Face tracking does nothing
Tick Track faces; the cascade must load from Code/Client/.
Detection feels slow
It runs on your computer. Improve lighting and face the camera.
Check two things

Is the server running on the Pi, and is the IP right? That fixes most connection problems.
